QuickSort(low,high:integer)
	var i,j,p;
	i←low;
	j←high;
	p←key[low];
	while low<high do
    	while low<high and key[high]>=p do
        	high--;
		if low<high do
        	key[low]←key[high];
			low++;
		else
       		break;
		while low<high and key[low]<=p do
        	low++;
		if low<high
        	key[high]←key[low];
			high--;
	key[low]←p;
	if i<low-1 do
        QuickSort(i,low-1);
	if high+1<j do
        QuickSort(high+1,j);

SelectSort(n:integer)

	for i←1 to n do
        k←i;
		for j←i+1 to n do
            if key[k]>key[j] do
                k←j;
		if k!=i do
            temp←key[k];
			key[k]←key[i];

    var i,j,temp:integer;
	flag:boolean;
	flag←true;
	for i←1 to n and flag do
        flag←false;
		for j←1 to n-i+1 do
            if key[j]>key[j+1] do
                temp←key[j+1];
				key[j+1]←key[j];
				key[j]←temp;
